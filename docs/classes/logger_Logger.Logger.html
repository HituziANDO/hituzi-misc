Class Logger

A logger class.

Example

Logger.default.level = LogLevel.Debug;

// Outputs debug logs.
Logger.default.debug("Hello, world!")();

// Outputs info logs.
Logger.default.info("Hello, world!")();

// Outputs warn logs.
Logger.default.warn("Hello, world!")();

// Outputs error logs.
Logger.default.error("Hello, world!")();

// Outputs no logs.
Logger.default.level = LogLevel.Off;
